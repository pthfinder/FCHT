tournament_update_stats_root_lost = {
	if = {
		limit = {
			exists = scope:opponent
		}
		scope:opponent = {
			if = {
				limit = {
					has_variable = number_of_captures
				}
				change_variable = {
					name = number_of_captures
					add = 1
				}
			}
			else = {
				set_variable = {
					name = number_of_captures
					value = 1
				}
			}

			if = {
				limit = {
					has_variable = gold_earned
				}
				change_variable = { 
					name = gold_earned
					add = root.ransom_value
				}
			}
			else = {
				set_variable = {
					name = gold_earned
					value = root.ransom_value
				}
			}
		}
	}
	if = {
		limit = {
			exists = root
		}
		root = {									
			if = {
				limit ={
					has_variable = number_of_times_captured
				}
				change_variable = {
					name = number_of_times_captured
					add = 1
				}
			}
			else = {
				set_variable = {
					name = number_of_times_captured
					value = 1
				}
			}

			if = {
				limit = {
					has_variable = gold_earned
				}
				change_variable = { 
					name = gold_earned
					subtract = root.ransom_value
				}
			}
			else = {
				set_variable = {
					name = gold_earned
					value = 0
				}
				change_variable = {
					name = gold_earned
					subtract = root.ransom_value
				}
			}
		}
	}
}
